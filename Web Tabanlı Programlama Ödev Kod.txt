function cikar()
{
	var sayi1 = document.getElementById("sayi1").value;
    var sayi2 = document.getElementById("sayi2").value;

    document.getElementById("sonucHTML").innerHTML= sonuc;

function carp()
{
	var sayi1 = document.getElementById("sayi1").value;
    var sayi2 = document.getElementById("sayi2").value;

    document.getElementById("sonucHTML").innerHTML= sonuc;

function bol()
{
	var sayi1 = document.getElementById("sayi1").value;
    var sayi2 = document.getElementById("sayi2").value;

    document.getElementById("sonucHTML").innerHTML= sonuc;

{
	document.getElementById("sayi1").value=0;
    document.getElementById("sayi2").value=0;
